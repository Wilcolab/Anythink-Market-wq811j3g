// Function to convert text to snake case
function toSnakeCase(text) {
  return text
    .replace(/\s+/g, ' ')
    .trim()
    .replace(/([a-z])([A-Z])/g, '$1_$2')
    .replace(/[\s\-]+/g, '_')
    .toLowerCase();
}

// Example usage
console.log(toSnakeCase('Hello World Example')); // Output: hello_world_example

module.exports = { toSnakeCase };
